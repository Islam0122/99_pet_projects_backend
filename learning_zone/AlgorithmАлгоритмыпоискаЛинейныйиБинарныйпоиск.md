def binary_search_iterative(arr, target):
    """
    –ë–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫ (–∏—Ç–µ—Ä–∞—Ç–∏–≤–Ω–∞—è –≤–µ—Ä—Å–∏—è).

    return -1
